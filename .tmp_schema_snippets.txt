model PlannedSession {
  id                   String                         @id @default(uuid()) @db.Uuid
  user_id              String                         @db.Uuid
  program_id           String                         @db.Uuid
  block_id             String?                        @db.Uuid
  session_template_id  String?                        @db.Uuid
  sequence_index       Int
  session_date         DateTime                       @db.Date
  status               SessionState
  generation_reason    PlannedSessionGenerationReason
  source_session_id    String?                        @db.Uuid
  planned_start_at     DateTime?                      @db.Timestamptz(6)
  planned_duration_min Int?
  objective_summary    String?                        @db.Text
  adaptation_snapshot  Json?
  notes                String?                        @db.Text
  created_at           DateTime                       @default(now()) @db.Timestamptz(6)
  updated_at           DateTime                       @updatedAt @db.Timestamptz(6)

  program            Program            @relation(fields: [program_id], references: [id])
  block              Block?             @relation(fields: [block_id], references: [id])
  session_template   SessionTemplate?   @relation(fields: [session_template_id], references: [id])
  source_session     PlannedSession?    @relation("PlannedSessionSource", fields: [source_session_id], references: [id])
  derived_sessions   PlannedSession[]   @relation("PlannedSessionSource")
  planned_units      PlannedUnit[]
  session_executions SessionExecution[]

  @@unique([program_id, sequence_index])
  @@index([user_id, session_date])
  @@index([program_id])
  @@index([block_id])
  @@index([session_template_id])
  @@index([source_session_id])
}
